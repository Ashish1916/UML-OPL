\section{Small-Step Semantics (25 points)}

Let us define the calculator language below that performs arithmetic and boolean operations. It has three different syntactic classes: (1) arithmetic expressions $a$ (2) boolean expressions $b$, and (3) final values $v$.

The following questions require you define small-step semantics. That is, first you need to define the configuration. Then, you need to write inference rules that show one configuration small-steps to another configuration.

\begin{align*}
n &\in \mathbb{Z} \\
a &::= n \mid a_1 + a_2 \mid a_1 \times a_2 \\
b &::= \text{true} \mid \text{false} \mid a = a \mid a \neq a \\
  &\quad \mid a \leq a \mid a > a \mid \neg b \mid b \&\& b \\
v &::= n \mid \text{true} \mid \text{false}
\end{align*}

\begin{enumerate}[label=(\alph*)]
\item Write small-step semantics for the syntactic class of arithmetic expressions generated by $a$.
\item Write small-step semantics for the syntactic class of boolean expressions generated by $b$.
\end{enumerate}

\subsection*{(a) Small-step semantics for arithmetic expressions}

Configuration: $\langle a, \sigma \rangle$, where $a$ is an arithmetic expression and $\sigma$ is the store.

Inference rules:

\begin{prooftree}
\AxiomC{}
\UnaryInfC{$\langle n, \sigma \rangle \rightarrow_a n$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\langle a_1, \sigma \rangle \rightarrow_a a_1'$}
\UnaryInfC{$\langle a_1 + a_2, \sigma \rangle \rightarrow_a a_1' + a_2$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\langle a_2, \sigma \rangle \rightarrow_a a_2'$}
\UnaryInfC{$\langle n_1 + a_2, \sigma \rangle \rightarrow_a n_1 + a_2'$}
\end{prooftree}

\begin{prooftree}
\AxiomC{}
\UnaryInfC{$\langle n_1 + n_2, \sigma \rangle \rightarrow_a n_3$}
\end{prooftree}
where $n_3$ is the sum of $n_1$ and $n_2$

\begin{prooftree}
\AxiomC{$\langle a_1, \sigma \rangle \rightarrow_a a_1'$}
\UnaryInfC{$\langle a_1 \times a_2, \sigma \rangle \rightarrow_a a_1' \times a_2$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\langle a_2, \sigma \rangle \rightarrow_a a_2'$}
\UnaryInfC{$\langle n_1 \times a_2, \sigma \rangle \rightarrow_a n_1 \times a_2'$}
\end{prooftree}

\begin{prooftree}
\AxiomC{}
\UnaryInfC{$\langle n_1 \times n_2, \sigma \rangle \rightarrow_a n_3$}
\end{prooftree}
where $n_3$ is the product of $n_1$ and $n_2$

\subsection*{(b) Small-step semantics for boolean expressions}

Configuration: $\langle b, \sigma \rangle$, where $b$ is a boolean expression and $\sigma$ is the store.

Inference rules:

\begin{prooftree}
\AxiomC{}
\UnaryInfC{$\langle \text{true}, \sigma \rangle \rightarrow_b \text{true}$}
\end{prooftree}

\begin{prooftree}
\AxiomC{}
\UnaryInfC{$\langle \text{false}, \sigma \rangle \rightarrow_b \text{false}$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\langle a_1, \sigma \rangle \rightarrow_a a_1'$}
\UnaryInfC{$\langle a_1 = a_2, \sigma \rangle \rightarrow_b a_1' = a_2$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\langle a_2, \sigma \rangle \rightarrow_a a_2'$}
\UnaryInfC{$\langle n_1 = a_2, \sigma \rangle \rightarrow_b n_1 = a_2'$}
\end{prooftree}

\begin{prooftree}
\AxiomC{}
\UnaryInfC{$\langle n_1 = n_2, \sigma \rangle \rightarrow_b \text{true}$}
\end{prooftree}
if $n_1 = n_2$

\begin{prooftree}
\AxiomC{}
\UnaryInfC{$\langle n_1 = n_2, \sigma \rangle \rightarrow_b \text{false}$}
\end{prooftree}
if $n_1 \neq n_2$

\begin{prooftree}
\AxiomC{$\langle a_1, \sigma \rangle \rightarrow_a a_1'$}
\UnaryInfC{$\langle a_1 \neq a_2, \sigma \rangle \rightarrow_b a_1' \neq a_2$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\langle a_2, \sigma \rangle \rightarrow_a a_2'$}
\UnaryInfC{$\langle n_1 \neq a_2, \sigma \rangle \rightarrow_b n_1 \neq a_2'$}
\end{prooftree}

\begin{prooftree}
\AxiomC{}
\UnaryInfC{$\langle n_1 \neq n_2, \sigma \rangle \rightarrow_b \text{true}$}
\end{prooftree}
if $n_1 \neq n_2$

\begin{prooftree}
\AxiomC{}
\UnaryInfC{$\langle n_1 \neq n_2, \sigma \rangle \rightarrow_b \text{false}$}
\end{prooftree}
if $n_1 = n_2$

\begin{prooftree}
\AxiomC{$\langle a_1, \sigma \rangle \rightarrow_a a_1'$}
\UnaryInfC{$\langle a_1 \leq a_2, \sigma \rangle \rightarrow_b a_1' \leq a_2$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\langle a_2, \sigma \rangle \rightarrow_a a_2'$}
\UnaryInfC{$\langle n_1 \leq a_2, \sigma \rangle \rightarrow_b n_1 \leq a_2'$}
\end{prooftree}

\begin{prooftree}
\AxiomC{}
\UnaryInfC{$\langle n_1 \leq n_2, \sigma \rangle \rightarrow_b \text{true}$}
\end{prooftree}
if $n_1 \leq n_2$

\begin{prooftree}
\AxiomC{}
\UnaryInfC{$\langle n_1 \leq n_2, \sigma \rangle \rightarrow_b \text{false}$}
\end{prooftree}
if $n_1 > n_2$

\begin{prooftree}
\AxiomC{$\langle a_1, \sigma \rangle \rightarrow_a a_1'$}
\UnaryInfC{$\langle a_1 > a_2, \sigma \rangle \rightarrow_b a_1' > a_2$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\langle a_2, \sigma \rangle \rightarrow_a a_2'$}
\UnaryInfC{$\langle n_1 > a_2, \sigma \rangle \rightarrow_b n_1 > a_2'$}
\end{prooftree}

\begin{prooftree}
\AxiomC{}
\UnaryInfC{$\langle n_1 > n_2, \sigma \rangle \rightarrow_b \text{true}$}
\end{prooftree}
if $n_1 > n_2$

\begin{prooftree}
\AxiomC{}
\UnaryInfC{$\langle n_1 > n_2, \sigma \rangle \rightarrow_b \text{false}$}
\end{prooftree}
if $n_1 \leq n_2$

\begin{prooftree}
\AxiomC{$\langle b, \sigma \rangle \rightarrow_b b'$}
\UnaryInfC{$\langle \neg b, \sigma \rangle \rightarrow_b \neg b'$}
\end{prooftree}

\begin{prooftree}
\AxiomC{}
\UnaryInfC{$\langle \neg \text{true}, \sigma \rangle \rightarrow_b \text{false}$}
\end{prooftree}

\begin{prooftree}
\AxiomC{}
\UnaryInfC{$\langle \neg \text{false}, \sigma \rangle \rightarrow_b \text{true}$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\langle b_1, \sigma \rangle \rightarrow_b b_1'$}
\UnaryInfC{$\langle b_1 \&\& b_2, \sigma \rangle \rightarrow_b b_1' \&\& b_2$}
\end{prooftree}

\begin{prooftree}
\AxiomC{}
\UnaryInfC{$\langle \text{true} \&\& b, \sigma \rangle \rightarrow_b b$}
\end{prooftree}

\begin{prooftree}
\AxiomC{}
\UnaryInfC{$\langle \text{false} \&\& b, \sigma \rangle \rightarrow_b \text{false}$}
\end{prooftree}
